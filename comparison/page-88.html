defaultpen(fontsize(10)); unitsize(1.5 cm); real angleindegrees(pair A, pair B, pair C) { real a, b, c; a = abs(B - C); b = abs(C - A); c = abs(A - B); return(aCos((a^2 + c^2 - b^2)/(2*a*c))); }; pair A, B, C, D, E, P, X; pair[] O; real angleA, angleB, angleC, x; A = (1,3); B = (0,0); C = (4,0); x = 50; angleA = angleindegrees(C,A,B); angleB = angleindegrees(A,B,C); angleC = angleindegrees(B,C,A); O[3] = extension(A, rotate(-(x + angleC - 90),A)*(B), B, rotate(x + angleC - 90,B)*(A)); O[2] = extension(A, rotate(x + angleB - 90,A)*(C), C, rotate(-(x + angleB - 90),C)*(A)); P = reflect(O[2],O[3])*(A); D = incenter(A,B,P); E = incenter(A,C,P); O[0] = circumcenter(A,B,C); X = O[0] + reflect(A,P)*(O[0]) - A; draw(circumcircle(A,B,C),red); draw(A--B--C--cycle); draw(B--P--C); draw(B--X--C); draw(A--X); label("$A$", A, N); label("$B$", B, SW); label("$C$", C, SE); label("$P$", P, NE); label("$X$", X, S);
