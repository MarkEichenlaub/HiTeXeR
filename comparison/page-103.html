// Draw hatch marks on the back of mirror int nHatch = 12; real hatchLen = 0.3; for(int i = 0; i <= nHatch; ++i) { real t = i/nHatch; pair p = point(mirrorArc, t * length(mirrorArc)); pair outward = unit(p - CoC); // Label C (label orientation stays normal) dot(CoC, black+linewidth(8)); label("$C$", CoC, W, fontsize(20)); // Mark vertex dot(vertex, black+linewidth(3)); // Ray from C to a point on the mirror real rayAngle = 20; pair mirrorPoint_orig = CoC_orig + R * dir(270 + rayAngle); pair mirrorPoint = rot * mirrorPoint_orig; // Rotation angle for reflected rays real theta_rot = 20; // degrees // Red ray: rotate about mirrorPoint by theta_rot clockwise pair redDir = unit(CoC - mirrorPoint); pair redDirRotated = rotate(-theta_rot) * redDir; pair redEnd = mirrorPoint + length(CoC - mirrorPoint) * redDirRotated; draw(redEnd -- mirrorPoint, red+linewidth(2.4), Arrow(size=8)); // Blue ray: rotate about mirrorPoint by theta_rot CCW pair blueDir = unit(CoC - mirrorPoint); pair blueDirRotated = rotate(theta_rot) * blueDir; pair blueEnd = mirrorPoint + length(CoC - mirrorPoint) * blueDirRotated; draw(mirrorPoint -- blueEnd, blue+linewidth(1.2), Arrow(size=8)); // Find focus F: intersection of blue ray with optical axis (x=0) pair blueRayDir = unit(blueEnd - mirrorPoint); real tF = -mirrorPoint.x / blueRayDir.x; pair F = mirrorPoint + tF * blueRayDir; dot(F, black+linewidth(8)); label("$F$", F, SW, fontsize(20)); // Mark point of incidence M dot(mirrorPoint, black+linewidth(8)); real arcStart = min(angleAxisDeg, angleNormalDeg); real arcEnd = max(angleAxisDeg, angleNormalDeg); draw(arc(CoC, arcRadius, arcStart, arcEnd), black+linewidth(0.8)); pair labelDir = dir(0.5*(arcStart + arcEnd)); label("$\theta$", CoC + (arcRadius + 0.35)*labelDir, fontsize(16)); // Angle mark at mirrorPoint between the normal (toward CoC) and the blue ray (angle of reflection) real angleNormalAtP = degrees(CoC - mirrorPoint);
