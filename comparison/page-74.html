size(350); real R = 100; real halfangle = 20; // Apply 90 degree clockwise rotation to all geometry transform T = rotate(-90); // Mirror arc (concave, opening downward after rotation) draw(arc((0,0), R, -halfangle-90, halfangle-90), linewidth(1.2)); // Dotted line for the rest of the circle draw(arc((0,0), R, halfangle-90, 360-halfangle-90), dotted); // Center point dot((0,0), linewidth(6)); label("$C$", (0,0), N, fontsize(18)); // Vertex is the bottommost point of the mirror arc after rotation pair V = T * (R * dir(0)); // Dashed line from C to vertex, labeled R draw((0,0)--V, dashed); label("$R$", midpoint((0,0)--V), W, fontsize(18)); // Label "spherical mirror" below the solid arc label("spherical concave mirror", T * (R * dir(0) + (5,0)), S, fontsize(18)); // Hash marks on the outside of the mirror arc int n = 7; real hashlen = 5; for (int i = 0; i <= n; ++i) { real angle = -halfangle + i * (2 * halfangle) / n; pair p = R * dir(angle); pair outward = dir(angle); pair tangent = rotate(90) * outward; pair tickdir = unit(outward + tangent); draw(T * p -- T * (p + hashlen * tickdir), linewidth(0.8)); } // Label "imaginary complete sphere" rotated position label(minipage("\centering imaginary\\complete\\sphere", 80), T * (1.2*R * dir(135))+(-20,0), fontsize(18));
