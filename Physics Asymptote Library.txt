How to Use
Modifying a diagram from the library
Use the table of contents to navigate and previews to decide if a pre-made diagram works for you. Click the “source” link.


Copy the code. Hit the “TeXeR” button in the top left, in the teal menu bar, wait for the page to load, and hit it again. This should give you a fresh window with a unique URL.


Paste the diagram, then modify it as needed.


Adding a diagram to the library


Make sure you have edit privileges. Mark E. can give them to you (eichenlaub@aops.com).


Go to the TeXeR (https://artofproblemsolving.com/texer). Optionally, choose a descriptive URL by typing it in place of the random letters in the address bar. E.g. go to https://artofproblemsolving.com/texer/EMWaves3D if you are making a 3D drawing of electromagnetic waves. 


Make your diagram in the TeXeR. Save it. Click “Make Public” on the far right of the top teal bar.

Copy an image of the output. In this doc, find the appropriate section, or make a new section. Title your drawing. Include the link to the TeXeR, your name as owner, and any other information you think people would want.

When you later use your own drawing, copy it, get a fresh TeXeR, and paste it there instead of editing the original, unless you are specifically updating the library entry, not just adapting the library entry for a particular use. This prevents you from accidentally changing the original.

//Atwood with n horizontal boxes, k vertical boxes, slopes phi, theta, masses m1, m2.


[asy]
unitsize(1.0 cm);
defaultpen(fontsize(12pt));


real x = 0, y = 0, hboxes = 2, vboxes = 2;
real hmass = 1, vmass = 1;
real hb = 1*hmass, vb = 1*vmass, theta = 0, phi = 10; //set phi = 90 left side for vertical atwood, angles > 25 deg require tweaking.
real hdist = 0.3/(Cos(theta)); //adjust for spacing between right boxes
real vdist = 0.6/(Cos(phi)+1); //adjust for spacing between left boxes
real vyshift = phi/200; //adjust to place left box on ramp surface
real hyshift = theta/220;  //adjust to place right box on ramp surface
pair A, B, C, D; //box corners
pair Z, X, Y, M, N, F, G; //right triangle, right box, left triangle
pair O = (0, 0);
//ground
real L = 5*hboxes*hb*Cos(theta), LL = 5*vboxes*vb*Cos(phi); //triangle hyp


//right side
real groundsize = 2.5; //extra vertical size of ground
Y = (0,-hb);
Z = (0, Y.y -L*Sin(theta));
X = (L*Cos(theta),Y.y-L*Sin(theta));
M = Z - (0, groundsize*vb*vboxes*Cos(theta)); 
N = (X.x,M.y);
filldraw(Z--X--Y--cycle,gray);
filldraw(M--N--X--Z--cycle,gray); 
draw(Z--X,gray+1bp);


//pulley
real r = hb/2;
real d; 
if (phi == 90){
        d = vb;
    } else {
    d = 1/4;
}
real alpha = atan((2*d-r)/r);
real l = (2*d - r)/Cos(alpha);
pair R = (-2*d+r, -r + l*Sin(theta));


filldraw(circle(R,r),gray);
draw(R+(r*Sin(theta),r*Cos(theta))--O,black+1bp);
draw(R--(0,-hb),black+2bp);


//left side
G = (0, Y.y - LL*Sin(phi));
F = (-LL*Cos(phi), Y.y - LL*Sin(phi));
pair K = (F.x, M.y);
filldraw(G--F--Y--cycle,gray);
filldraw(G--F--K--M--cycle,gray); 
draw(F--G,gray+1bp);
draw(O -(0,hb)--M,gray+1bp);
path box;


//hboxes
x = 2;


draw(((x-hdist*hb)*Cos(theta),-(x-hdist*hb)*Sin(theta))--(R.x + r*Sin(theta),R.y +r*Cos(theta)),black+1pt);
for (int i = 0; i < hboxes; i = i+1){
    draw(((x-hdist*hb)*Cos(theta),-(x-hdist*hb)*Sin(theta))--((x+hdist*hb)*Cos(theta),-(x+hdist*hb)*Sin(theta)),black+1pt);
    x = x + (hb+hdist)*Cos(theta);
    A = (x - hb, y + hyshift - hb);
    B = (x + hb, y + hyshift - hb);
    C = (x + hb, y + hyshift + hb);
    D = (x - hb, y + hyshift + hb);
    box = A--B--C--D--cycle;
    filldraw(rotate(-theta)*box,lightgray);
    x = x + (hb+hdist)*Cos(theta);
}


//vboxes


if (phi == 90){
    x = R.x - r;
    y = R.y - 3*vb;
    draw((x,y)--(x, R.y),black+1bp);
    for (int i = 0; i < vboxes; i = i+1){
        draw((x,y - vb)--(x,y),black+1bp);
        y = y - 2*vb;
        A = (x - vb, y - vb);
        B = (x + vb, y - vb);
        C = (x + vb, y + vb);
        D = (x - vb, y + vb);
        y = y - vb;
        filldraw(A--B--C--D--cycle, lightgray);
    }
} else {
    x = -2;
    y = 0;
    draw(((x+vdist*vb*vmass)*Cos(phi), (x+vdist*vb*vmass)*Sin(phi)+vyshift)--(R.x - r*Sin(phi),R.y + r*Cos(phi)),black+1pt);
    for (int i = 0; i < vboxes; i = i+1){
        draw(((x-vdist*vb)*Cos(phi),(x-vdist*hb)*Sin(phi)+vyshift)--((x+vdist*hb)*Cos(phi),(x+vdist*vb)*Sin(phi)+vyshift),black+1pt);
        x = x - (vb+vdist)*Cos(phi);
        A = (x - vb, y + vyshift - vb);
        B = (x + vb, y + vyshift - vb);
        C = (x + vb, y + vyshift + vb);
        D = (x - vb, y + vyshift + vb);
        box = A--B--C--D--cycle;
        filldraw(rotate(phi)*box,lightgray);
        x = x - (vb+vdist)*Cos(phi);
    }
}


[/asy]

[asy]
unitsize(2 cm);
defaultpen(fontsize(14pt));
//L = length of slope
//l = distance between object and foot of slope
//theta = angle of slope
//F = length of force vector
real L = 5.5, l = 2.5, theta = 30, F = 2.25;


pair A, B, C, X, Y, W, Z, M, N;


A = (L*Cos(theta),0);
B = (0,0);
C = (0,L*Sin(theta));
X = interp(A,C,l/L);
Y = X + (0,-F);
Z = (Y + reflect(A,C)*(Y))/2;
W = X + Y - Z;
M = interp(A,C,(1.4*l)/L) +(0,0.1);
N = interp(A,C,(1.2*l)/L) +(0,-0.1);


filldraw(A--B--C--cycle,gray,gray);

//theta = Angle of axis (relative to vertical)
//D = diameter of spring
//L = length of spring
//N = number of coils
//r = ratio of major/minor axes in ellipse
//A = location of spring bottom
real theta = 20;
real D = 0.25;
real L = 1;
real N = 15;
real r = 0.5;
pair A = (0,0);


real spacing = L/N;
real y = spacing*Cos(theta);
real x = spacing*Sin(theta);


for (int i = 0; i<N; ++i) { 
  pair center = A + (i*x,i*y); 
  path E = rotate(-theta,center)*ellipse(center,D,r*D); 
  draw(E);
}

//theta = angle of slope

//A = foot of slope
//B = slope vertex with right angle
//C = peak of slope
//X = Box corner lowest on slope
//W, X, Y, Z = Box corners (listed in clockwise order) 
pair A, B, C, X, Y, W, Z;


real x = s*Sin(theta);
real y = s*Cos(theta);

B = (0,0);
C = (0,L*Sin(theta));

//theta = Angle of axis (relative to vertical)
//D = diameter of spring
//L = length of spring
//N = number of coils
//r = ratio of major/minor axes in ellipse
//A = location of spring bottom
real theta = 0;
real D = 0.25;
real L = 1;
real N = 15;
real r = 0.5;
pair A = (0,0);


real spacing = L/N;
real y = spacing*Cos(theta);
real x = spacing*Sin(theta);


for (int i = 0; i<N; ++i) { 
  pair center = A + (i*x,i*y); 
  path E = rotate(-theta,center)*ellipse(center,D,r*D); 
  draw(E);
}
